  final userId = client.userId().toString();

  }
  try {
    if (!await client.deactivate(password)) {
      throw 'Deactivating the client failed';
    }
  } catch (e) {
    // reset the client locally
    _clients.add(client);
    _index = clients.length - 1;
    rethrow;

  await _persistSessions();
