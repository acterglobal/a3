Function konst::string::from_utf8

pub const fn from_utf8(slice: &[u8]) -> Result<&str, Utf8Error>
Expand description

Delegates to core::str::from_utf8, wrapping the error to provide a panic method for use in unwrap_ctx

§Example

§Basic

use konst::{
    result::unwrap_ctx,
    string,
};

const STR: &str = unwrap_ctx!(string::from_utf8(b"foo bar"));

assert_eq!(STR, "foo bar")
